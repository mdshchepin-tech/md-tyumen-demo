        function calculateQuizScore() {

            let score = 0;

            

            // Question 1

            const q1Answers = document.querySelectorAll('input[name="q1"]:checked');

            let q1Score = 0;

            q1Answers.forEach(answer => {

                q1Score += parseFloat(answer.value);

            });

            score += Math.min(q1Score, 1); // Cap at 1 point for question 1

            

            // Question 2

            const q2Answers = document.querySelectorAll('input[name="q2"]:checked');

            let q2Score = 0;

            q2Answers.forEach(answer => {

                q2Score += parseFloat(answer.value);

            });

            score += Math.min(q2Score, 1); // Cap at 1 point for question 2

            

            // Question 3

            const q3Answer = document.querySelector('input[name="q3"]:checked');

            if (q3Answer) {

                score += parseFloat(q3Answer.value);

            }

            

            totalScore += score;

            updateScoreDisplay();

            checkVictory();

            

            alert(`Вы набрали ${score} баллов в викторине! Общий счет: ${totalScore}`);

        }

        

        function calculateCaseScore() {

            // Stage 1

            const stage1Answer = document.querySelector('input[name="stage1"]:checked');

            if (stage1Answer) {

                score += parseFloat(stage1Answer.value);

            }

            

            // Stage 2

            const stage2Answer = document.querySelector('input[name="stage2"]:checked');

            if (stage2Answer) {

                score += parseFloat(stage2Answer.value);

            }

            

            // Stage 3

            const stage3Answer = document.querySelector('input[name="stage3"]:checked');

            if (stage3Answer) {

                score += parseFloat(stage3Answer.value);

            }

            

            // Stage 4

            const stage4Answer = document.querySelector('input[name="stage4"]:checked');

            if (stage4Answer) {

                score += parseFloat(stage4Answer.value);

            }

            

            // Stage 5

            const stage5Answers = document.querySelectorAll('input[name="stage5"]:checked');

            let stage5Score = 0;

            stage5Answers.forEach(answer => {

                stage5Score += parseFloat(answer.value);
